002/**
003 * This class is part of the "World of Zuul" application. 
004 * "World of Zuul" is a very simple, text based adventure game.  
005 * 
006 * This class holds an enumeration table of all command words known to the game.
007 * It is used to recognise commands as they are typed in.
008 *
009 * @author  Michael Kolling and David J. Barnes + D.Bureau
010 * @version 2008.03.30 + 2013.09.15
011 */
012public class CommandWords
013{
014    // tableau constant qui contient tous les mots de commande valides
015    private static final String[] sValidCommands = { "eat", "quit", "help", "look","go", "back", "test", "take", "drop" };
016
017    /**
018     * Constructeur par defaut
019     */
020    public CommandWords()
021    {
022        // rien a faire pour le moment...
023    } // CommandWords()
024
025    /**
026     * Check if the word is in the valid commands. 
027     * Return true if it is, else false
028     */
029    public boolean isCommand( final String pString )
030    {
031        // pour chacune des commandes valides (du tableau constant)
032        for ( int i=0; i<sValidCommands.length; i++ ) //i++ = i+1
033        {
034            // si elle est egale a pString
035            if ( sValidCommands[i].equals( pString ) )
036                return true;
037        } // for
038        // si nous arrivons la, c'est que la commande
039        // n'a pas ete trouvee dans le tableau
040        return false;
041    } // isCommand()
042
043    /**
044     * Return valid commands in a board
045     */
046    public String getCommandList()
047    {
048        StringBuilder vCommands = new StringBuilder();
049        for(int i = 0; i < sValidCommands.length; i++)
050            {
051                vCommands.append( sValidCommands[i] + "\n");
052            } 
053        return vCommands.toString();
054    }
055} // CommandWords
