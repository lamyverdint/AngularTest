001import javax.swing.*;
002import java.awt.*;
003import java.awt.event.*;
004import java.net.URL;
005import java.awt.image.*;
006
007import javax.swing.JTextField;
008import javax.swing.JTextArea;
009import javax.swing.JLabel;
010import javax.swing.JScrollPane;
011import javax.swing.JPanel;
012import javax.swing.JFrame;
013import javax.swing.JButton;
014
015/**
016 * This class implements a simple graphical user interface with a text entry
017 * area, a text output area and an optional image.
018 * 
019 * @author Michael Kolling (DB edited)
020 * @version 1.0 (Jan 2003)
021 */
022public class UserInterface implements ActionListener
023{
024    private GameEngine aEngine;
025    private JFrame     aMyFrame;
026    private JTextField aEntryField;
027    private JTextArea  aLog;
028    private JLabel     aImage;
029    private JButton    aButton;
030
031    /**
032     * Create a UserInterface
033     */
034    public UserInterface( final GameEngine pGameEngine )
035    {
036        this.aEngine = pGameEngine;
037        this.createGUI();
038    } // UserInterface(.)
039
040    /**
041     * Prints out some text into the text area.
042     */
043    public void print( final String pText )
044    {
045        this.aLog.append( pText );
046        this.aLog.setCaretPosition( this.aLog.getDocument().getLength() );
047    } // print(.)
048
049    /**
050     * Prints out some text into the text area, followed by a line break.
051     */
052    public void println( final String pText )
053    {
054        this.print( pText + "\n" );
055    } // println(.)
056
057    /**
058     * Show an image file in the interface.
059     */
060    public void showImage( final String pImageName )
061    {
062        URL vImageURL = this.getClass().getClassLoader().getResource( pImageName );
063        if ( vImageURL == null )
064            System.out.println( "image not found" );
065        else {
066            ImageIcon vIcon = new ImageIcon( vImageURL );
067            this.aImage.setIcon( vIcon );
068            this.aMyFrame.pack();
069        }
070    } // showImage(.)
071
072    /**
073     * Enable or disable input in the input field.
074     */
075    public void enable( final boolean pOnOff )
076    {
077        this.aEntryField.setEditable( pOnOff );
078        if ( ! pOnOff )
079            this.aEntryField.getCaret().setBlinkRate( 0 );
080    } // enable(.)
081
082    /**
083     * Set up graphical user interface.
084     */
085    private void createGUI()
086    {
087        this.aMyFrame = new JFrame( "The Crazy Nest" );
088        this.aEntryField = new JTextField( 34 );
089
090        this.aLog = new JTextArea();
091        this.aLog.setEditable( false );
092        JScrollPane vListScroller = new JScrollPane( this.aLog );
093        vListScroller.setPreferredSize( new Dimension(200, 200) );
094        vListScroller.setMinimumSize( new Dimension(100,100) );
095
096        JPanel vPanel = new JPanel();
097        this.aImage = new JLabel();
098        this.aButton = new JButton("Back");
099
100        vPanel.setLayout( new BorderLayout() );
101        vPanel.add( this.aImage, BorderLayout.NORTH );
102        vPanel.add( vListScroller, BorderLayout.CENTER );
103        vPanel.add( this.aEntryField, BorderLayout.SOUTH );
104        vPanel.add( this.aButton, BorderLayout.EAST );
105        //vPanel.add(this.aButtonS, BorderLayout.EAST);
106
107        this.aMyFrame.getContentPane().add( vPanel, BorderLayout.CENTER );
108
109        // add some event listeners to some components
110        this.aMyFrame.addWindowListener( new WindowAdapter() {
111            public void windowClosing(WindowEvent e) { System.exit(0); }
112        } );
113
114        this.aEntryField.addActionListener( this );
115        this.aButton.addActionListener( this );
116        //this.aButtonS.addActionListener( this );
117
118        this.aMyFrame.pack();
119        this.aMyFrame.setVisible( true );
120        this.aEntryField.requestFocus();
121    } // createGUI()
122
123    
124    /**
125     * Actionlistener interface for entry textfield.
126     */    
127    public void actionPerformed( final ActionEvent pE ) 
128    {
129        if(pE.getSource() == this.aButton)
130            aEngine.interpretCommand("back");
131            else
132            this.processCommand();
133    } // actionPerformed(.)
134
135    /**
136     * A command has been entered. Read the command and do whatever is 
137     * necessary to process it.
138     */
139    private void processCommand()
140    {
141        String vInput = this.aEntryField.getText();
142        this.aEntryField.setText( "" );
143
144        this.aEngine.interpretCommand( vInput );
145    } // processCommand()
146} // UserInterface
